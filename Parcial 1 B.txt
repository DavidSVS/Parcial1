CREATE SEQUENCE DELITOS_SEQ
INCREMENT BY 1
START WITH 100
MAXVALUE 200
MINVALUE 100
NOCYCLE
NOORDER;

CREATE SEQUENCE CAPTURAS_SEQ
INCREMENT BY 1
START WITH 1000
MAXVALUE 2000
MINVALUE 1000
NOCYCLE
NOORDER;

CREATE TABLE DELITOS(
  ID integer PRIMARY KEY,
  NOMBRE VARCHAR2(255)
);

CREATE TABLE CAPTURAS(
  ID integer PRIMARY KEY,
  FECHA timestamp,
  DEPARTAMENTO VARCHAR(255),
  MUNICIPIO VARCHAR(255),
  DIA VARCHAR(255),
  BARRIO VARCHAR(255),
  ZONA VARCHAR(255),
  CLASE_SITIOEDAD VARCHAR(255), 
  SEXO VARCHAR(255), 
  ESTADO_CIVIL VARCHAR(255), 
  CLASE_EMPLEADO VARCHAR(255), 
  PROFESION VARCHAR(255), 
  ESCOLARIDAD VARCHAR(255) ,
  DELITO_ID integer,
  FOREIGN KEY (DELITO_ID) REFERENCES  DELITOS (ID) 
);
